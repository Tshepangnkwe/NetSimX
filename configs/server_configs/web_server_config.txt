## Apache Configuration (Linux)
### /etc/apache2/sites-available/netsimx.local.conf

# <VirtualHost *:80>
#     ServerName www.netsimx.local
#     ServerAlias netsimx.local
#     DocumentRoot /var/www/netsimx
#     ErrorLog ${APACHE_LOG_DIR}/netsimx_error.log
#     CustomLog ${APACHE_LOG_DIR}/netsimx_access.log combined
# 
#     <Directory /var/www/netsimx>
#         Options Indexes FollowSymLinks
#         AllowOverride All
#         Require all granted
#     </Directory>
# </VirtualHost>
